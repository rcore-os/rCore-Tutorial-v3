73    pub fn new(elf_data: &[u8]) -> Self {
74        // memory_set with elf program headers/trampoline/trap context/user stack
75        let (memory_set, user_sp, entry_point) = MemorySet::from_elf(elf_data);
76        let trap_cx_ppn = memory_set
77            .translate(VirtAddr::from(TRAP_CONTEXT).into())
78            .unwrap()
79            .ppn();
80        // alloc a pid and a kernel stack in kernel space
81        let pid_handle = pid_alloc();
82        let kernel_stack = KernelStack::new(&pid_handle);
83        let kernel_stack_top = kernel_stack.get_top();
84        let task_control_block = Self {
85            pid: pid_handle,
86            kernel_stack,
87            inner: unsafe {
88                UPSafeCell::new(TaskControlBlockInner {
89                    trap_cx_ppn,
90                    base_size: user_sp,
91                    task_cx: TaskContext::goto_trap_return(kernel_stack_top),
92                    task_status: TaskStatus::Ready,
93                    memory_set,
94                    parent: None,
95                    children: Vec::new(),
96                    exit_code: 0,
97                    fd_table: vec![
98                        // 0 -> stdin
99                        Some(Arc::new(Stdin)),
100                        // 1 -> stdout
101                        Some(Arc::new(Stdout)),
102                        // 2 -> stderr
103                        Some(Arc::new(Stdout)),
104                    ],
105                    signals: SignalFlags::empty(),
106                    signal_mask: SignalFlags::empty(),
107                    handling_sig: -1,
108                    signal_actions: SignalActions::default(),
109                    killed: false,
110                    frozen: false,
111                    trap_ctx_backup: None,
112                })
113            },
114        };
115        // prepare TrapContext in user space
116        let trap_cx = task_control_block.inner_exclusive_access().get_trap_cx();
117        *trap_cx = TrapContext::app_init_context(
118            entry_point,
119            user_sp,
120            KERNEL_SPACE.exclusive_access().token(),
121            kernel_stack_top,
122            trap_handler as usize,
123        );
124        task_control_block
125    }
126    pub fn exec(&self, elf_data: &[u8], args: Vec<String>) {
127        // memory_set with elf program headers/trampoline/trap context/user stack
128        let (memory_set, mut user_sp, entry_point) = MemorySet::from_elf(elf_data);
129        let trap_cx_ppn = memory_set
130            .translate(VirtAddr::from(TRAP_CONTEXT).into())
131            .unwrap()
132            .ppn();
133        // push arguments on user stack
134        user_sp -= (args.len() + 1) * core::mem::size_of::<usize>();
135        let argv_base = user_sp;
136        let mut argv: Vec<_> = (0..=args.len())
137            .map(|arg| {
138                translated_refmut(
139                    memory_set.token(),
140                    (argv_base + arg * core::mem::size_of::<usize>()) as *mut usize,
141                )
142            })
143            .collect();
144        *argv[args.len()] = 0;
145        for i in 0..args.len() {
146            user_sp -= args[i].len() + 1;
147            *argv[i] = user_sp;
148            let mut p = user_sp;
149            for c in args[i].as_bytes() {
150                *translated_refmut(memory_set.token(), p as *mut u8) = *c;
151                p += 1;
152            }
153            *translated_refmut(memory_set.token(), p as *mut u8) = 0;
154        }
155        // make the user_sp aligned to 8B for k210 platform
156        user_sp -= user_sp % core::mem::size_of::<usize>();
157
158        // **** access current TCB exclusively
159        let mut inner = self.inner_exclusive_access();
160        // substitute memory_set
161        inner.memory_set = memory_set;
162        // update trap_cx ppn
163        inner.trap_cx_ppn = trap_cx_ppn;
164        // initialize trap_cx
165        let mut trap_cx = TrapContext::app_init_context(
166            entry_point,
167            user_sp,
168            KERNEL_SPACE.exclusive_access().token(),
169            self.kernel_stack.get_top(),
170            trap_handler as usize,
171        );
172        trap_cx.x[10] = args.len();
173        trap_cx.x[11] = argv_base;
174        *inner.get_trap_cx() = trap_cx;
175        // **** release current PCB
176    }
177    pub fn fork(self: &Arc<TaskControlBlock>) -> Arc<TaskControlBlock> {
178        // ---- hold parent PCB lock
179        let mut parent_inner = self.inner_exclusive_access();
180        // copy user space(include trap context)
181        let memory_set = MemorySet::from_existed_user(&parent_inner.memory_set);
182        let trap_cx_ppn = memory_set
183            .translate(VirtAddr::from(TRAP_CONTEXT).into())
184            .unwrap()
185            .ppn();
186        // alloc a pid and a kernel stack in kernel space
187        let pid_handle = pid_alloc();
188        let kernel_stack = KernelStack::new(&pid_handle);
189        let kernel_stack_top = kernel_stack.get_top();
190        // copy fd table
191        let mut new_fd_table: Vec<Option<Arc<dyn File + Send + Sync>>> = Vec::new();
192        for fd in parent_inner.fd_table.iter() {
193            if let Some(file) = fd {
194                new_fd_table.push(Some(file.clone()));
195            } else {
196                new_fd_table.push(None);
197            }
198        }
199        let task_control_block = Arc::new(TaskControlBlock {
200            pid: pid_handle,
201            kernel_stack,
202            inner: unsafe {
203                UPSafeCell::new(TaskControlBlockInner {
204                    trap_cx_ppn,
205                    base_size: parent_inner.base_size,
206                    task_cx: TaskContext::goto_trap_return(kernel_stack_top),
207                    task_status: TaskStatus::Ready,
208                    memory_set,
209                    parent: Some(Arc::downgrade(self)),
210                    children: Vec::new(),
211                    exit_code: 0,
212                    fd_table: new_fd_table,
213                    signals: SignalFlags::empty(),
214                    // inherit the signal_mask and signal_action
215                    signal_mask: parent_inner.signal_mask,
216                    handling_sig: -1,
217                    signal_actions: parent_inner.signal_actions.clone(),
218                    killed: false,
219                    frozen: false,
220                    trap_ctx_backup: None,
221                })
222            },
223        });
224        // add child
225        parent_inner.children.push(task_control_block.clone());
226        // modify kernel_sp in trap_cx
227        // **** access child PCB exclusively
228        let trap_cx = task_control_block.inner_exclusive_access().get_trap_cx();
229        trap_cx.kernel_sp = kernel_stack_top;
230        // return
231        task_control_block
232        // **** release child PCB
233        // ---- release parent PCB
234    }
