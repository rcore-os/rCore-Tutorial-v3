42impl SignalFlags {
43    pub fn check_error(&self) -> Option<(i32, &'static str)> {
44        if self.contains(Self::SIGINT) {
45            Some((-2, "Killed, SIGINT=2"))
46        } else if self.contains(Self::SIGILL) {
47            Some((-4, "Illegal Instruction, SIGILL=4"))
48        } else if self.contains(Self::SIGABRT) {
49            Some((-6, "Aborted, SIGABRT=6"))
50        } else if self.contains(Self::SIGFPE) {
51            Some((-8, "Erroneous Arithmetic Operation, SIGFPE=8"))
52        } else if self.contains(Self::SIGKILL) {
53            Some((-9, "Killed, SIGKILL=9"))
54        } else if self.contains(Self::SIGSEGV) {
55            Some((-11, "Segmentation Fault, SIGSEGV=11"))
56        } else {
57            //println!("[K] signalflags check_error  {:?}", self);
58            None
