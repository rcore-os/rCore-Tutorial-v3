38fn clear_bss() {
39    unsafe extern "C" {
40        safe fn sbss();
41        safe fn ebss();
42    }
43    unsafe {
44        core::slice::from_raw_parts_mut(sbss as usize as *mut u8, ebss as usize - sbss as usize)
45            .fill(0);
46    }
47}
