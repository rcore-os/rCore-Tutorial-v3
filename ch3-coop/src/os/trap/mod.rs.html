1//! Trap handling functionality
2//!
3//! For rCore, we have a single trap entry point, namely `__alltraps`. At
4//! initialization in [`init()`], we set the `stvec` CSR to point to it.
5//!
6//! All traps go through `__alltraps`, which is defined in `trap.S`. The
7//! assembly language code does just enough work restore the kernel space
8//! context, ensuring that Rust code safely runs, and transfers control to
9//! [`trap_handler()`].
10//!
11//! It then calls different functionality based on what exactly the exception
12//! was. For example, timer interrupts trigger task preemption, and syscalls go
13//! to [`syscall()`].
14
15mod context;
16
17use crate::syscall::syscall;
18use core::arch::global_asm;
19use riscv::register::{
20    mtvec::TrapMode,
21    scause::{self, Exception, Trap},
22    stval, stvec,
23};
24
25global_asm!(include_str!("trap.S"));
26
27/// initialize CSR `stvec` as the entry of `__alltraps`
28pub fn init() {
29    unsafe extern "C" {
30        safe fn __alltraps();
31    }
32    unsafe {
33        stvec::write(__alltraps as usize, TrapMode::Direct);
34    }
35}
36
37#[unsafe(no_mangle)]
38/// handle an interrupt, exception, or system call from user space
39pub fn trap_handler(cx: &mut TrapContext) -> &mut TrapContext {
40    let scause = scause::read(); // get trap cause
41    let stval = stval::read(); // get extra value
42    match scause.cause() {
43        Trap::Exception(Exception::UserEnvCall) => {
44            cx.sepc += 4;
45            cx.x[10] = syscall(cx.x[17], [cx.x[10], cx.x[11], cx.x[12]]) as usize;
46        }
47        Trap::Exception(Exception::StoreFault) | Trap::Exception(Exception::StorePageFault) => {
48            println!(
49                "[kernel] PageFault in application, bad addr = {:#x}, bad instruction = {:#x}, kernel killed it.",
50                stval, cx.sepc
51            );
52            panic!("[kernel] Cannot continue!");
53            //run_next_app();
54        }
55        Trap::Exception(Exception::IllegalInstruction) => {
56            println!("[kernel] IllegalInstruction in application, kernel killed it.");
57            panic!("[kernel] Cannot continue!");
58            //run_next_app();
59        }
60        _ => {
61            panic!(
62                "Unsupported trap {:?}, stval = {:#x}!",
63                scause.cause(),
64                stval
65            );
66        }
67    }
68    cx
69}
